# Set node positions explicitly 
pos={'andre':(0,1),
     'beverly':(0,-1),
     'carol':(1,2),
     'diane':(1,0),
     'ed':(1,-2),
     'ike':(4,0),
     'garth':(2,-1),
     'jane':(5,0),
     'fernando':(2,1),
     'heather':(3,0)}

# nodes

# edges
nx.draw_networkx_edges(G,pos,alpha=0.5,width=6)

# labels
nx.draw_networkx_labels(G,pos,font_size=15,font_family='sans-serif')

plt.axis('off')
